CREATE TABLE "correlations" (

"id" bigint PRIMARY KEY GENERATED ALWAYS AS IDENTITY (sequence name "correlations_id_seq" INCREMENT BY 1 MINVALUE 1 MAXVALUE 9223372036854775807 START WITH 1 CACHE 1),

"industry_id" bigint NOT NULL UNIQUE,

"indicator_id" bigint NOT NULL UNIQUE,

"correlation" double precision NOT NULL,

"window_days" integer NOT NULL,

"n_observations" integer NOT NULL,

"computed_at" timestamp with time zone DEFAULT now() NOT NULL,

CONSTRAINT "correlations_industry_id_indicator_id_key" UNIQUE("industry_id","indicator_id"),

CONSTRAINT "correlations_correlation_check" CHECK (((correlation >= ('-1'::integer)::double precision) AND (correlation <= (1)::double precision)))

);

CREATE TABLE "index_observations" (

"id" bigint PRIMARY KEY GENERATED ALWAYS AS IDENTITY (sequence name "index_observations_id_seq" INCREMENT BY 1 MINVALUE 1 MAXVALUE 9223372036854775807 START WITH 1 CACHE 1),

"industry_id" bigint NOT NULL UNIQUE,

"observation_date" date NOT NULL UNIQUE,

"value" numeric(20, 6) NOT NULL,

"created_at" timestamp with time zone DEFAULT now() NOT NULL,

CONSTRAINT "index_observations_industry_id_observation_date_key" UNIQUE("industry_id","observation_date")

);

CREATE TABLE "indicator_observations" (

"id" bigint PRIMARY KEY GENERATED ALWAYS AS IDENTITY (sequence name "indicator_observations_id_seq" INCREMENT BY 1 MINVALUE 1 MAXVALUE 9223372036854775807 START WITH 1 CACHE 1),

"indicator_id" bigint NOT NULL UNIQUE,

"observation_date" date NOT NULL UNIQUE,

"value" numeric(20, 6) NOT NULL,

"created_at" timestamp with time zone DEFAULT now() NOT NULL,

CONSTRAINT "indicator_observations_indicator_id_observation_date_key" UNIQUE("indicator_id","observation_date")

);

CREATE TABLE "indicators" (

"id" bigint PRIMARY KEY GENERATED ALWAYS AS IDENTITY (sequence name "indicators_id_seq" INCREMENT BY 1 MINVALUE 1 MAXVALUE 9223372036854775807 START WITH 1 CACHE 1),

"code" text NOT NULL UNIQUE,

"name" text NOT NULL,

"source" text NOT NULL UNIQUE,

"unit" text NOT NULL,

"frequency" text NOT NULL,

"transform" text DEFAULT 'yoy_pct' NOT NULL,

"norm_lookback_days" integer DEFAULT 1825 NOT NULL,

"is_active" boolean DEFAULT true NOT NULL,

"created_at" timestamp with time zone DEFAULT now() NOT NULL,

CONSTRAINT "indicators_source_code_key" UNIQUE("source","code"),

CONSTRAINT "indicators_frequency_check" CHECK ((frequency = ANY (ARRAY['daily'::text, 'weekly'::text, 'monthly'::text, 'quarterly'::text]))),

CONSTRAINT "indicators_transform_check" CHECK ((transform = ANY (ARRAY['level'::text, 'mom_change'::text, 'yoy_change'::text, 'mom_pct'::text, 'yoy_pct'::text])))

);

CREATE TABLE "industries" (

"id" bigint PRIMARY KEY GENERATED ALWAYS AS IDENTITY (sequence name "industries_id_seq" INCREMENT BY 1 MINVALUE 1 MAXVALUE 9223372036854775807 START WITH 1 CACHE 1),

"code" text NOT NULL CONSTRAINT "industries_code_key" UNIQUE,

"name" text NOT NULL,

"index_proxy" text,

"parent_id" bigint,

"created_at" timestamp with time zone DEFAULT now() NOT NULL

);

CREATE TABLE "securities" (

"id" bigint PRIMARY KEY GENERATED ALWAYS AS IDENTITY (sequence name "securities_id_seq" INCREMENT BY 1 MINVALUE 1 MAXVALUE 9223372036854775807 START WITH 1 CACHE 1),

"ticker" text NOT NULL UNIQUE,

"name" text NOT NULL,

"exchange" text NOT NULL UNIQUE,

"currency" char(3) DEFAULT 'USD' NOT NULL,

"industry_id" bigint NOT NULL,

"is_active" boolean DEFAULT true NOT NULL,

"created_at" timestamp with time zone DEFAULT now() NOT NULL,

CONSTRAINT "securities_ticker_exchange_key" UNIQUE("ticker","exchange")

);
